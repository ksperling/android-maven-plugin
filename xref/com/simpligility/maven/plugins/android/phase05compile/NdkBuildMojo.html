297     /**
298      * @throws MojoExecutionException
299      * @throws MojoFailureException
300      */
301     public void execute() throws MojoExecutionException, MojoFailureException
302     {
303         // Validate the NDK
304         final File ndkBuildFile = new File( getAndroidNdk().getNdkBuildPath() );
305         NativeHelper.validateNDKVersion( ndkBuildFile.getParentFile() );
306 
307         validateMakefile( project, makefile );
308 
309         final String[] resolvedNDKArchitectures = NativeHelper.getNdkArchitectures(
310             ndkArchitecture != null ? ndkArchitecture : ndkArchitectures, applicationMakefile,
311             project.getBasedir() );
312 
313         for ( String architecture : resolvedNDKArchitectures )
314         {
315             try
316             {
317                 compileForArchitecture( architecture );
318             }
319             catch ( IOException e )
320             {
321                 getLog().error( "Error while executing: " + e.getMessage() );
322                 throw new MojoExecutionException( e.getMessage(), e );
323             }
324             catch ( ExecutionException e )
325             {
326                 getLog().error( "Error while executing: " + e.getMessage() );
327                 throw new MojoExecutionException( e.getMessage(), e );
328             }
329         }
330     }

332     private void compileForArchitecture( String architecture )
333         throws MojoExecutionException, IOException, ExecutionException
334     {
335         getLog().debug( "Resolving for NDK architecture : " + architecture );
336 
337         // Start setting up the command line to be executed
338         final CommandExecutor executor = CommandExecutor.Factory.createDefaultCommmandExecutor();
339         // Add an error listener to the build - this allows the build to conditionally fail
340         // depending on a) the output of the build b) whether or not build errors (output on stderr) should be
341         // ignored and c) whether the pattern matches or not
342         executor.setErrorListener( getNdkErrorListener() );
343 
344         final Set<Artifact> nativeLibraryArtifacts = findNativeLibraryDependencies();
345 
346         // If there are any static libraries the code needs to link to, include those in the make file
347         final Set<Artifact> resolveNativeLibraryArtifacts =
348             getArtifactResolverHelper().resolveArtifacts( nativeLibraryArtifacts );
349 
350         getLog().debug( "resolveArtifacts found " + resolveNativeLibraryArtifacts.size()
351             + ": " + resolveNativeLibraryArtifacts.toString() );
352 
353         final File buildFolder = new File( ndkBuildDirectory, architecture );
354         buildFolder.mkdirs();
355 
356         final File androidMavenMakefile = new File( buildFolder, "android_maven_plugin_makefile.mk" );
357         final MakefileHelper makefileHelper = new MakefileHelper( getLog(),
358             getArtifactResolverHelper(),
359             harArtifactHandler, getUnpackedLibsDirectory()
360         );
361 
362         final MakefileHelper.MakefileHolder makefileHolder = makefileHelper
363             .createMakefileFromArtifacts( resolveNativeLibraryArtifacts, architecture, "armeabi", useHeaderArchives );
364 
365         final FileOutputStream output = new FileOutputStream( androidMavenMakefile );
366         try
367         {
368             IOUtil.copy( makefileHolder.getMakeFile(), output );
369         }
370         finally
371         {
372             output.close();
373         }
374 
375         // Add the path to the generated makefile - this is picked up by the build (by an include from the user)
376         executor.addEnvironment( "ANDROID_MAVEN_PLUGIN_MAKEFILE", androidMavenMakefile.getAbsolutePath() );
377 
378         setupNativeLibraryEnvironment( makefileHelper, executor, resolveNativeLibraryArtifacts,
379             architecture );
380 
381         // Adds the location of the Makefile capturer file - this file will after the build include
382         // things like header files, flags etc.  It is processed after the build to retrieve the headers
383         // and also capture flags etc ...
384         final File makefileCaptureFile = File.createTempFile( "android_maven_plugin_makefile_captures", ".tmp" );
385         makefileCaptureFile.deleteOnExit();
386         executor.addEnvironment( MakefileHelper.MAKEFILE_CAPTURE_FILE, makefileCaptureFile.getAbsolutePath() );
387 
388         // Add any defined system properties
389         if ( systemProperties != null && ! systemProperties.isEmpty() )
390         {
391             for ( Map.Entry<String, String> entry : systemProperties.entrySet() )
392             {
393                 executor.addEnvironment( entry.getKey(), entry.getValue() );
394             }
395         }
396         executor.setLogger( this.getLog() );
397         // Setup the command line for the make
398         final List<String> commands = new ArrayList<String>();
399         // Setup the build directory (defaults to the current directory) but may be different depending
400         // on user configuration
401         commands.add( "-C" );
402         commands.add( project.getBasedir().getAbsolutePath() );
403 
404         // If the build should use a custom makefile or not - some validation is done to ensure
405         // this exists and all
406         if ( makefile != null )
407         {
408             File makeFile = new File( project.getBasedir(), makefile );
409             if ( ! makeFile.exists() )
410             {
411                 getLog().error( "Specified makefile " + makeFile + " does not exist" );
412                 throw new MojoExecutionException( "Specified makefile " + makeFile + " does not exist" );
413             }
414             commands.add( "-f" );
415             commands.add( makefile );
416         }
417 
418         configureApplicationMakefile( commands );
419         configureMaxJobs( commands );
420         configureNdkToolchain( architecture, commands );
421         configureAdditionalCommands( commands );
422 
423         // If a build target is specified, tag that onto the command line as the very last of the parameters
424         if ( target != null )
425         {
426             commands.add( target );
427         }
428         else /*if ( Const.ArtifactType.NATIVE_IMPLEMENTATION_ARCHIVE.equals( project.getPackaging() ) )*/
429         {
430             commands.add( project.getArtifactId() );
431         }
432 
433         final String ndkBuildPath = resolveNdkBuildExecutable();
434         getLog().debug( ndkBuildPath + " " + commands.toString() );
435         getLog().info( "Executing NDK " + architecture + " make at : " + ndkBuildDirectory );
436 
437         executor.setCaptureStdOut( true );
438         executor.executeCommand( ndkBuildPath, commands, ndkBuildDirectory, true );
439         getLog().debug( "Executed NDK " + architecture + " make at : " + ndkBuildDirectory );
440 
441         // Where the NDK build creates the libs.
442         final File nativeLibOutputDirectory = new File( nativeLibrariesOutputDirectory, architecture );
443         nativeLibOutputDirectory.mkdirs();
444 
445         // Move the built native libs into the packaging folder.
446         // We don't create them there to start with because the NDK build seems determined to create them in the root.
447         final File destinationDirectory = new File( ndkOutputDirectory, architecture );
448         FileUtils.moveDirectory( nativeLibOutputDirectory, destinationDirectory );
449 
450         // Attempt to attach the native library if the project is defined as a "pure" native Android library
451         // (packaging is 'so' or 'a') or if the plugin has been configured to attach the native library to the build
452         if ( Const.ArtifactType.NATIVE_SYMBOL_OBJECT.equals( project.getPackaging() )
453             || Const.ArtifactType.NATIVE_IMPLEMENTATION_ARCHIVE.equals( project.getPackaging() )
454             || attachNativeArtifacts )
455         {
456             attachNativeLib( destinationDirectory, architecture );
457         }
458 
459         // Process conditionally any of the headers to include into the header archive file
460         if ( attachHeaderFiles )
461         {
462             attachHeaderFiles( makefileCaptureFile, architecture );
463         }
464 
465         // If we created a makefile for the build we should be polite and remove any extracted include
466         // directories after we're done
467         getLog().info( "Cleaning up extracted include directories used for build" );
468         MakefileHelper.cleanupAfterBuild( makefileHolder );
469     }

508     private void configureNdkToolchain( String architecture, List<String> commands )
509             throws MojoExecutionException
510     {
511         if ( ndkToolchain != null )
512         {
513             // Setup the correct toolchain to use
514             // FIXME: perform a validation that this toolchain exists in the NDK
515             commands.add( "NDK_TOOLCHAIN=" + ndkToolchain );
516         }
517         else
518         {
519             // Resolve the toolchain from the architecture
520             // <ndkArchitectures>
521             //   <x86>x86-4.6</x86>
522             //   <armeabi>x86-4.6</armeabi>
523             //   <arm64-v8a>aarch64-4.9</arm64-v8a>
524             // </ndkArchitectures>
525             final String toolchainFromArchitecture = getAndroidNdk().getToolchainFromArchitecture(
526                     architecture, ndkArchitectureToolchainMappings );
527             getLog().debug( "Resolved toolchain for " + architecture + " to " + toolchainFromArchitecture );
528             commands.add( "NDK_TOOLCHAIN=" + toolchainFromArchitecture );
529             commands.add( "APP_ABI=" + architecture );
530 
531         }
532     }

534     /**
535      * Attaches native libs to project.
536      */
537     private void attachNativeLib( File destinationDirectory, String architecture )
538             throws IOException, MojoExecutionException
539     {
540         final File nativeArtifactFile;
541         if ( ndkFinalLibraryName == null )
542         {
543             nativeArtifactFile = findNativeLibrary( destinationDirectory );
544         }
545         else
546         {
547             nativeArtifactFile = nativeLibraryFromName( destinationDirectory );
548         }
549 
550         final String artifactType = resolveArtifactType( nativeArtifactFile );
551         if ( nativeArtifactFile.getName().endsWith( ".so" ) && ! skipStripping )
552         {
553             getLog().debug( "Post processing (stripping) native compiled artifact: " + nativeArtifactFile );
554             invokeNDKStripper( nativeArtifactFile );
555         }
556 
557         getLog().debug( "Adding native compiled artifact: " + nativeArtifactFile );
558 
559         final String classifier = ( ndkClassifier == null )
560                 ? architecture
561                 : architecture + "-" + ndkClassifier;
562 
563         projectHelper.attachArtifact( this.project, artifactType, classifier, nativeArtifactFile );
564     }

566     /**
567      * Search the specified directory for native artifacts that match the artifact Id
568      */
569     private File findNativeLibrary( File nativeLibDirectory ) throws MojoExecutionException
570     {
571         getLog().info( "Searching " + nativeLibDirectory + " for built library" );
572         File[] files = nativeLibDirectory.listFiles( new FilenameFilter()
573         {
574             public boolean accept( final File dir, final String name )
575             {
576                 String libraryName = ndkFinalLibraryName;
577 
578                 if ( libraryName == null || libraryName.isEmpty() )
579                 {
580                     libraryName = project.getArtifactId();
581                 }
582 
583                 // FIXME: The following logic won't work for an APKLIB building a static library
584                 if ( Const.ArtifactType.NATIVE_IMPLEMENTATION_ARCHIVE.equals( project.getPackaging() ) )
585                 {
586                     return name.startsWith(
587                             "lib" + libraryName ) && name.endsWith( ".a" );
588                 }
589                 else
590                 {
591                     return name.startsWith(
592                             "lib" + libraryName ) && name.endsWith( ".so" );
593                 }
594             }
595         } );
596         // slight limitation at this stage - we only handle a single .so artifact
597         if ( files == null || files.length != 1 )
598         {
599             getLog().warn( "Error while detecting native compile artifacts: "
600                     + ( files == null || files.length == 0
601                             ? "None found"
602                             : "Found more than 1 artifact" ) );
603             if ( target != null )
604             {
605                 getLog().warn( "Using the 'target' configuration option to specify the output file name "
606                         + "is no longer supported, use 'ndkFinalLibraryName' instead." );
607             }
608 
609             if ( files != null && files.length > 1 )
610             {
611                 getLog().debug( "List of files found: " + Arrays.asList( files ) );
612                 getLog().error(
613                         "Currently, only a single, final native library is supported by the build" );
614                 throw new MojoExecutionException(
615                         "Currently, only a single, final native library is supported by the build" );
616             }
617             else
618             {
619                 getLog().error( "No native compiled library found, did the native compile complete "
620                         + "successfully?" );
621                 throw new MojoExecutionException( "No native compiled library found, did the native "
622                         + "compile complete successfully?" );
623             }
624         }
625         return files[ 0 ];
626     }

628     private File nativeLibraryFromName( File nativeLibDirectory ) throws MojoExecutionException
629     {
630         final File libraryFile;
631         // Find the nativeArtifactFile in the nativeLibDirectory/ndkFinalLibraryName
632         if ( Const.ArtifactType.NATIVE_SYMBOL_OBJECT.equals( project.getPackaging() )
633                 || Const.ArtifactType.NATIVE_IMPLEMENTATION_ARCHIVE.equals( project.getPackaging() ) )
634         {
635             libraryFile = new File( nativeLibDirectory,
636                     "lib" + ndkFinalLibraryName + "." + project.getPackaging() );
637         }
638         else
639         {
640             final File staticLib = new File( nativeLibDirectory,
641                     "lib" + ndkFinalLibraryName + ".a" );
642             if ( staticLib.exists() )
643             {
644                 libraryFile = staticLib;
645             }
646             else
647             {
648                 libraryFile = new File( nativeLibDirectory,
649                         "lib" + ndkFinalLibraryName + ".so" );
650             }
651         }
652         if ( ! libraryFile.exists() )
653         {
654             getLog().error(
655                     "Could not locate final native library using the provided ndkFinalLibraryName "
656                             + ndkFinalLibraryName + " (tried " + libraryFile.getAbsolutePath()
657                             + ")" );
658             throw new MojoExecutionException(
659                     "Could not locate final native library using the provided ndkFinalLibraryName "
660                     + ndkFinalLibraryName + " (tried " + libraryFile.getAbsolutePath()
661                     + ")" );
662         }
663 
664         return libraryFile;
665     }

667     private CommandExecutor.ErrorListener getNdkErrorListener()
668     {
669         return new CommandExecutor.ErrorListener()
670         {
671             @Override
672             public boolean isError( String error )
673             {
674 
675                 // Unconditionally ignore *All* build warning if configured to
676                 if ( ignoreBuildWarnings )
677                 {
678                     return false;
679                 }
680 
681                 final Pattern pattern = Pattern.compile( buildWarningsRegularExpression );
682                 final Matcher matcher = pattern.matcher( error );
683 
684                 // If the the reg.exp actually matches, we can safely say this is not an error
685                 // since in theory the user told us so
686                 if ( matcher.matches() )
687                 {
688                     return false;
689                 }
690 
691                 // Otherwise, it is just another error
692                 return true;
693             }
694         };
695     }

707     private void invokeNDKStripper( File file ) throws MojoExecutionException
708     {
709         try
710         {
711             getLog().debug( "Detected shared library artifact, will now strip it" );
712             // Execute the strip command
713             final CommandExecutor stripCommandExecutor = CommandExecutor.Factory.createDefaultCommmandExecutor();
714             stripCommandExecutor.setErrorListener( new CommandExecutor.ErrorListener()
715             {
716                 public boolean isError( String error )
717                 {
718                     getLog().error( "Error while stripping binary: " + error );
719                     return true;
720                 }
721             } );
722             stripCommandExecutor.setLogger( getLog() );
723             stripCommandExecutor.setCaptureStdOut( true );
724             stripCommandExecutor.executeCommand( resolveNdkStripper( file ).getAbsolutePath(),
725                                                  Arrays.asList( file.getAbsolutePath() ) );
726         }
727         catch ( ExecutionException e )
728         {
729             getLog().error( "Error while attempting to strip shared library", e );
730             throw new MojoExecutionException( "Error while attempting to strip shared library" );
731         }
732     }

756     private void attachHeaderFiles( File localCIncludesFile, String architecture )
757         throws MojoExecutionException, IOException
758     {
759 
760         final List<HeaderFilesDirective> finalHeaderFilesDirectives = new ArrayList<HeaderFilesDirective>();
761 
762         if ( useLocalSrcIncludePaths )
763         {
764             Properties props = new Properties();
765             props.load( new FileInputStream( localCIncludesFile ) );
766             String localCIncludes = props.getProperty( "LOCAL_C_INCLUDES" );
767             if ( localCIncludes != null && ! localCIncludes.trim().isEmpty() )
768             {
769                 String[] includes = localCIncludes.split( " " );
770                 for ( String include : includes )
771                 {
772                     final HeaderFilesDirective headerFilesDirective = new HeaderFilesDirective();
773                     File includeDir = new File( project.getBasedir(), include );
774                     headerFilesDirective.setDirectory( includeDir.getAbsolutePath() );
775                     headerFilesDirective.setIncludes( new String[]{ "**/*.h" } );
776                     finalHeaderFilesDirectives.add( headerFilesDirective );
777                 }
778             }
779         }
780         else
781         {
782             if ( headerFilesDirectives != null )
783             {
784                 finalHeaderFilesDirectives.addAll( headerFilesDirectives );
785             }
786         }
787         if ( finalHeaderFilesDirectives.isEmpty() )
788         {
789             getLog().debug( "No header files included, will add default set" );
790             final HeaderFilesDirective e = new HeaderFilesDirective();
791             e.setDirectory( new File( project.getBasedir() + "/jni" ).getAbsolutePath() );
792             e.setIncludes( new String[]{ "**/*.h" } );
793             finalHeaderFilesDirectives.add( e );
794         }
795         createHeaderArchive( finalHeaderFilesDirectives, architecture );
796     }

798     private void createHeaderArchive( List<HeaderFilesDirective> finalHeaderFilesDirectives, String architecture )
799             throws MojoExecutionException
800     {
801         try
802         {
803             MavenArchiver mavenArchiver = new MavenArchiver();
804             mavenArchiver.setArchiver( jarArchiver );
805 
806             final File jarFile = new File( targetDirectory, finalName + ".har" );
807 
808             mavenArchiver.setOutputFile( jarFile );
809 
810             for ( HeaderFilesDirective headerFilesDirective : finalHeaderFilesDirectives )
811             {
812                 mavenArchiver.getArchiver().addDirectory( new File( headerFilesDirective.getDirectory() ),
813                         headerFilesDirective.getIncludes(), headerFilesDirective.getExcludes() );
814             }
815 
816             final MavenArchiveConfiguration mavenArchiveConfiguration = new MavenArchiveConfiguration();
817             mavenArchiveConfiguration.setAddMavenDescriptor( false );
818 
819             mavenArchiver.createArchive( project, mavenArchiveConfiguration );
820 
821             String classifier = architecture;
822             if ( ndkClassifier != null )
823             {
824                 classifier += "-" + ndkClassifier;
825             }
826 
827             getLog().debug( "Attaching 'har' classifier=" + classifier + " file=" + jarFile );
828             projectHelper.attachArtifact( project, Const.ArtifactType.NATIVE_HEADER_ARCHIVE, classifier, jarFile );
829 
830         }
831         catch ( Exception e )
832         {
833             throw new MojoExecutionException( e.getMessage() );
834         }
835     }

863     private Set<Artifact> findNativeLibraryDependencies() throws MojoExecutionException
864     {
865         final NativeHelper nativeHelper = getNativeHelper();
866         final Set<Artifact> staticLibraryArtifacts = nativeHelper
867                 .getNativeDependenciesArtifacts( this, getUnpackedLibsDirectory(), false );
868         final Set<Artifact> sharedLibraryArtifacts = nativeHelper
869                 .getNativeDependenciesArtifacts( this, getUnpackedLibsDirectory(), true );
870 
871         final Set<Artifact> mergedArtifacts = new LinkedHashSet<Artifact>();
872         filterNativeDependencies( mergedArtifacts, staticLibraryArtifacts );
873         filterNativeDependencies( mergedArtifacts, sharedLibraryArtifacts );
874 
875         getLog().debug( "findNativeLibraryDependencies found " + mergedArtifacts.size()
876                 + ": " + mergedArtifacts.toString() );
877 
878         return mergedArtifacts;
879     }

881     /**
882      * Selectively add artifacts from source to target excluding any whose groupId and artifactId match
883      * the current build.
884      *
885      * Introduced to work around an issue when the ndk-build is executed twice by maven for example when
886      * invoking maven 'install site'. In this case the artifacts attached by the first invocation are
887      * found but are not valid dependencies and must be excluded.
888      *
889      * @param targetSet artifact Set to copy in to
890      * @param source artifact Set to filter
891      */
892     private void filterNativeDependencies( Set<Artifact> targetSet, Set<Artifact> source )
893     {
894         for ( Artifact a : source )
895         {
896             if ( project.getGroupId().equals( a.getGroupId() )
897                     && project.getArtifactId().equals( a.getArtifactId() ) )
898             {
899                 getLog().warn( "Excluding native dependency attached by this build" );
900             }
901             else
902             {
903                 targetSet.add( a );
904             }
905         }
906     }

908     /**
909      * Resolve the artifact type from the current project and the specified file.  If the project packaging is
910      * either 'a' or 'so' it will use the packaging, otherwise it checks the file for the extension
911      *
912      * @param file The file being added as an artifact
913      * @return The artifact type (so or a)
914      */
915     private String resolveArtifactType( File file )
916     {
917         if ( Const.ArtifactType.NATIVE_SYMBOL_OBJECT.equals( project.getPackaging() )
918                 || Const.ArtifactType.NATIVE_IMPLEMENTATION_ARCHIVE.equals( project.getPackaging() ) )
919         {
920             return project.getPackaging();
921         }
922         else
923         {
924             // At this point, the file (as found by our filtering previously will end with either 'so' or 'a'
925             return file.getName().endsWith( Const.ArtifactType.NATIVE_SYMBOL_OBJECT )
926                     ? Const.ArtifactType.NATIVE_SYMBOL_OBJECT
927                     : Const.ArtifactType.NATIVE_IMPLEMENTATION_ARCHIVE;
928         }
929     }
