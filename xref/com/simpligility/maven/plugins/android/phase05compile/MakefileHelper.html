38      /**
39       * Holder for the result of creating a makefile.  This in particular keep tracks of all directories created
40       * for extracted header files.
41       */
42      public static class MakefileHolder
43      {
44          String makeFile;
45          List<File> includeDirectories;
46  
47          public MakefileHolder( List<File> includeDirectories, String makeFile )
48          {
49              this.includeDirectories = includeDirectories;
50              this.makeFile = makeFile;
51          }
52  
53          public List<File> getIncludeDirectories()
54          {
55              return includeDirectories;
56          }
57  
58          public String getMakeFile()
59          {
60              return makeFile;
61          }
62      }

85      /**
86       * Cleans up all include directories created in the temp directory during the build.
87       *
88       * @param makefileHolder The holder produced by the
89       * {@link MakefileHelper#createMakefileFromArtifacts(Set, String, String, boolean)}
90       */
91      public static void cleanupAfterBuild( MakefileHolder makefileHolder )
92      {
93  
94          if ( makefileHolder.getIncludeDirectories() != null )
95          {
96              for ( File file : makefileHolder.getIncludeDirectories() )
97              {
98                  try
99                  {
100                     FileUtils.deleteDirectory( file );
101                 }
102                 catch ( IOException e )
103                 {
104                     e.printStackTrace();
105                 }
106             }
107         }
108 
109     }

111     /**
112      * Creates an Android Makefile based on the specified set of static library dependency artifacts.
113      *
114      * @param artifacts         The list of (static library) dependency artifacts to create the Makefile from
115      * @param useHeaderArchives If true, the Makefile should include a LOCAL_EXPORT_C_INCLUDES statement, pointing to
116      *                          the location where the header archive was expanded
117      * @return The created Makefile
118      */
119     public MakefileHolder createMakefileFromArtifacts( Set<Artifact> artifacts,
120                                                               String ndkArchitecture, String defaultNDKArchitecture,
121                                                               boolean useHeaderArchives )
122             throws IOException, MojoExecutionException
123     {
124 
125         final StringBuilder makeFile = new StringBuilder( "# Generated by Android Maven Plugin\n" );
126         final List<File> includeDirectories = new ArrayList<File>();
127 
128         // Add now output - allows us to somewhat intelligently determine the include paths to use for the header
129         // archive
130         makeFile.append( "$(shell echo \"LOCAL_C_INCLUDES=$(LOCAL_C_INCLUDES)\" > $(" + MAKEFILE_CAPTURE_FILE + "))" );
131         makeFile.append( '\n' );
132         makeFile.append( "$(shell echo \"LOCAL_PATH=$(LOCAL_PATH)\" >> $(" + MAKEFILE_CAPTURE_FILE + "))" );
133         makeFile.append( '\n' );
134         makeFile.append( "$(shell echo \"LOCAL_MODULE_FILENAME=$(LOCAL_MODULE_FILENAME)\" >> $("
135                 + MAKEFILE_CAPTURE_FILE + "))" );
136         makeFile.append( '\n' );
137         makeFile.append( "$(shell echo \"LOCAL_MODULE=$(LOCAL_MODULE)\" >> $(" + MAKEFILE_CAPTURE_FILE + "))" );
138         makeFile.append( '\n' );
139         makeFile.append( "$(shell echo \"LOCAL_CFLAGS=$(LOCAL_CFLAGS)\" >> $(" + MAKEFILE_CAPTURE_FILE + "))" );
140         makeFile.append( '\n' );
141 
142         if ( ! artifacts.isEmpty() )
143         {
144             for ( Artifact artifact : artifacts )
145             {
146                 final String architecture = NativeHelper.extractArchitectureFromArtifact( artifact,
147                         defaultNDKArchitecture );
148 
149                 makeFile.append( '\n' );
150                 makeFile.append( "ifeq ($(TARGET_ARCH_ABI)," ).append( architecture ).append( ")\n" );
151 
152                 makeFile.append( "#\n" );
153                 makeFile.append( "# Group ID: " );
154                 makeFile.append( artifact.getGroupId() );
155                 makeFile.append( '\n' );
156                 makeFile.append( "# Artifact ID: " );
157                 makeFile.append( artifact.getArtifactId() );
158                 makeFile.append( '\n' );
159                 makeFile.append( "# Artifact Type: " );
160                 makeFile.append( artifact.getType() );
161                 makeFile.append( '\n' );
162                 makeFile.append( "# Version: " );
163                 makeFile.append( artifact.getVersion() );
164                 makeFile.append( '\n' );
165                 makeFile.append( "include $(CLEAR_VARS)" );
166                 makeFile.append( '\n' );
167                 makeFile.append( "LOCAL_MODULE    := " );
168                 makeFile.append( artifact.getArtifactId() );
169                 makeFile.append( '\n' );
170 
171                 final boolean apklibStatic = addLibraryDetails( makeFile, artifact, ndkArchitecture );
172 
173                 if ( useHeaderArchives )
174                 {
175                     try
176                     {
177                         // Fix for dealing with APKLIBs - unfortunately it does not fully work since
178                         // an APKLIB can contain any number of architectures making it somewhat to resolve the
179                         // related (HAR) artifact.
180                         //
181                         // In this case, we construct the classifier from <architecture> and the artifact classifier
182                         // if it is also present.  Only issue is that if the APKLIB contains more than the armeabi
183                         // libraries (e.g. x86 for examples) the HAR is not resolved correctly.
184                         //
185                         String classifier = artifact.getClassifier();
186                         if ( "apklib".equals( artifact.getType() ) )
187                         {
188                             classifier = ndkArchitecture;
189                             if ( artifact.getClassifier() != null )
190                             {
191                                 classifier += "-" + artifact.getClassifier();
192                             }
193                         }
194 
195                         Artifact harArtifact = new DefaultArtifact( artifact.getGroupId(), artifact.getArtifactId(),
196                                 artifact.getVersion(), artifact.getScope(),
197                                 Const.ArtifactType.NATIVE_HEADER_ARCHIVE, classifier,
198                                 harArtifactHandler );
199 
200                         File resolvedHarArtifactFile = artifactResolverHelper.resolveArtifactToFile( harArtifact );
201                         log.debug( "Resolved har artifact file : " + resolvedHarArtifactFile );
202 
203                         final File includeDir = new File( System.getProperty( "java.io.tmpdir" ),
204                                 "android_maven_plugin_native_includes" + System.currentTimeMillis() + "_"
205                                         + harArtifact.getArtifactId() );
206                         includeDir.deleteOnExit();
207                         includeDirectories.add( includeDir );
208 
209                         JarHelper.unjar( new JarFile( resolvedHarArtifactFile ), includeDir,
210                                 new JarHelper.UnjarListener()
211                                 {
212                                     @Override
213                                     public boolean include( JarEntry jarEntry )
214                                     {
215                                         return ! jarEntry.getName().startsWith( "META-INF" );
216                                     }
217                                 } );
218 
219                         makeFile.append( "LOCAL_EXPORT_C_INCLUDES := " );
220                         makeFile.append( includeDir.getAbsolutePath() );
221                         makeFile.append( '\n' );
222                         
223                         if ( log.isDebugEnabled() )
224                         {
225                             Collection<File> includes = FileUtils.listFiles( includeDir,
226                                     TrueFileFilter.INSTANCE, TrueFileFilter.INSTANCE );
227                             log.debug( "Listing LOCAL_EXPORT_C_INCLUDES for " + artifact.getId() + ": " + includes );
228                         }
229                     }
230                     catch ( RuntimeException e )
231                     {
232                         throw new MojoExecutionException(
233                                 "Error while resolving header archive file for: " + artifact.getArtifactId(), e );
234                     }
235                 }
236                 if ( Const.ArtifactType.NATIVE_IMPLEMENTATION_ARCHIVE.equals( artifact.getType() ) || apklibStatic )
237                 {
238                     makeFile.append( "include $(PREBUILT_STATIC_LIBRARY)\n" );
239                 }
240                 else
241                 {
242                     makeFile.append( "include $(PREBUILT_SHARED_LIBRARY)\n" );
243                 }
244 
245                 makeFile.append( "endif #" ).append( artifact.getClassifier() ).append( '\n' );
246                 makeFile.append( '\n' );
247 
248             }
249         }
250         
251         return new MakefileHolder( includeDirectories, makeFile.toString() );
252     }

254     private boolean addLibraryDetails( StringBuilder makeFile,
255                                        Artifact artifact, String ndkArchitecture ) throws IOException
256     {
257         boolean apklibStatic = false;
258         if ( AndroidExtension.APKLIB.equals( artifact.getType() ) )
259         {
260             String classifier = artifact.getClassifier();
261             String architecture = ( classifier != null ) ? classifier : ndkArchitecture;
262             // 
263             // We assume that APKLIB contains a single static OR shared library
264             // that we should link against. The follow code identifies that file.
265             //
266             File[] staticLibs = NativeHelper.listNativeFiles( artifact, unpackedApkLibsDirectory, 
267                                                               architecture, true );
268             if ( staticLibs != null && staticLibs.length > 0 )
269             {
270                 int libIdx = findApklibNativeLibrary( staticLibs, artifact.getArtifactId() );
271                 apklibStatic = true;
272                 addLibraryDetails( makeFile, staticLibs[libIdx], "" );
273             }
274             else
275             {
276                 File[] sharedLibs = NativeHelper.listNativeFiles( artifact, unpackedApkLibsDirectory, 
277                                                                   architecture, false );
278                 if ( sharedLibs == null )
279                 {
280                     throw new IOException( "Failed to find any library file in APKLIB" );
281                 }
282                 int libIdx = findApklibNativeLibrary( sharedLibs, artifact.getArtifactId() );
283                 addLibraryDetails( makeFile, sharedLibs[libIdx], "" );
284             }
285         }
286         else
287         {
288             addLibraryDetails( makeFile, artifact.getFile(), artifact.getArtifactId() );
289         }
290 
291         return apklibStatic;
292     }

294     private void addLibraryDetails( StringBuilder makeFile, File libFile, String outputName )
295         throws IOException
296     {
297         makeFile.append( "LOCAL_PATH := " );
298         makeFile.append( libFile.getParentFile().getAbsolutePath() );
299         makeFile.append( '\n' );
300         makeFile.append( "LOCAL_SRC_FILES := " );
301         makeFile.append( libFile.getName() );
302         makeFile.append( '\n' );
303         makeFile.append( "LOCAL_MODULE_FILENAME := " );
304         if ( "".equals( outputName ) )
305         {
306             makeFile.append( FilenameUtils.removeExtension( libFile.getName() ) );
307         }
308         else
309         {
310             makeFile.append( outputName );
311         }
312         makeFile.append( '\n' );
313     }

315     /**
316      * @param libs the array of possible library files. Must not be null.
317      * @return the index in the array of the library to use
318      * @throws IOException if a library cannot be identified
319      */
320     private int findApklibNativeLibrary( File[] libs, String artifactName ) throws IOException
321     {
322         int libIdx = -1;
323         
324         if ( libs.length == 1 )
325         
326         {
327             libIdx = 0;
328         }
329         else
330         {
331             log.info( "Found multiple library files, looking for name match with artifact" );
332             StringBuilder sb = new StringBuilder();
333             for ( int i = 0; i < libs.length; i++ )
334             {
335                 if ( sb.length() != 0 )
336                 {
337                     sb.append( ", " );
338                 }
339                 sb.append( libs[i].getName() );
340                 if ( libs[i].getName().startsWith( "lib" + artifactName ) )
341                 {
342                     if ( libIdx != -1 )
343                     {
344                         // We have multiple matches, tell the user we can't handle this ...
345                         throw new IOException( "Found multiple libraries matching artifact name " + artifactName
346                                 + ". Please use unique artifact/library names." );
347                         
348                     }
349                     libIdx = i;
350                 }
351             }
352             if ( libIdx < 0 )
353             {
354                 throw new IOException( "Unable to determine main library from " + sb.toString()
355                         + " APKLIB should contain only 1 library or a library matching the artifact name" );
356             }
357         }
358         return libIdx;
359     }

361     /**
362      * Creates a list of artifacts suitable for use in the LOCAL_STATIC_LIBRARIES or LOCAL_SHARED_LIBRARIES 
363      * variable in an Android makefile
364      *
365      * @param resolvedLibraryList
366      * @param staticLibrary
367      * @return a list of Ids for artifacts that include static or shared libraries
368      */
369     public String createLibraryList( Set<Artifact> resolvedLibraryList,
370                                      String ndkArchitecture,
371                                      boolean staticLibrary )
372     {
373         Set<String> libraryNames = new LinkedHashSet<String>();
374 
375         for ( Artifact a : resolvedLibraryList )
376         {
377             if ( staticLibrary && Const.ArtifactType.NATIVE_IMPLEMENTATION_ARCHIVE.equals( a.getType() ) )
378             {
379                 libraryNames.add( a.getArtifactId() );
380             }
381             if ( ! staticLibrary && Const.ArtifactType.NATIVE_SYMBOL_OBJECT.equals( a.getType() ) )
382             {
383                 libraryNames.add( a.getArtifactId() );
384             }
385             if ( AndroidExtension.APKLIB.equals( a.getType() ) || AndroidExtension.AAR.equals( a.getType() ) )
386             {
387                 File[] libFiles = NativeHelper.listNativeFiles( a, unpackedApkLibsDirectory, 
388                                                                 ndkArchitecture, staticLibrary );
389                 if ( libFiles != null && libFiles.length > 0 )
390                 {
391                     libraryNames.add( a.getArtifactId() );
392                 }
393                 
394             }
395         }
396 
397         StringBuilder sb = new StringBuilder();
398 
399         Iterator<String> iter = libraryNames.iterator();
400 
401         while ( iter.hasNext() )
402         {
403             sb.append( iter.next() );
404 
405             if ( iter.hasNext() )
406             {
407                 sb.append( " " );
408             }
409         }
410 
411         return sb.toString();
412     }
