22  @RunWith(MavenJUnitTestRunner.class)
23  @MavenVersions({"3.2.3"})
24  public class ExtractDuplicatesIT {
25  
26      @Rule
27      public final TestResources resources = new TestResources();
28  
29      public final MavenRuntime mavenRuntime;
30  
31      public ExtractDuplicatesIT(MavenRuntimeBuilder builder) throws Exception {
32          this.mavenRuntime = builder.withCliOptions( "-X" ).build();
33      }
34  
35      @Test
36      public void buildDeployAndRun() throws Exception {
37          File basedir = resources.getBasedir( "duplicates" );
38          MavenExecutionResult result = mavenRuntime
39                  .forProject(basedir)
40                  .execute( "clean",
41                          "install" );
42  
43          result.assertErrorFreeLog();
44          result.assertLogText( "Duplicate file resourceA" );
45          result.assertLogText( "Duplicate file resourceB" );
46          result.assertLogText( "Duplicate file resourceC" );
47          File apk = new File(result.getBasedir().getAbsolutePath()+"/duplicates-app/target", "duplicates-app.apk");
48          assertNotNull("APK Not Null", apk);
49          ZipFile apkFile = new ZipFile(apk);
50          assertNotNull(apkFile.getEntry("resourceA"));
51          assertNotNull(apkFile.getEntry("resourceB"));
52          assertNotNull(apkFile.getEntry("resourceC"));
53  
54          //test services
55          assertEntryContents(apkFile, "META-INF/services/com.jayway.maven.plugins.android.TestInterface",
56                  "ImplementationA\nImplementationB\nImplementationC\nImplementationApp");
57  
58          //test xpath xml
59          assertEntryContents( apkFile, "META-INF/kmodule.xml", expectedKmodule );
60          assertEntryContents( apkFile, "kmodule.info", expectedInfo );
61      }
62  
63      private void assertEntryContents( ZipFile zip, String name, String expected ) throws IOException {
64          ZipEntry ze = zip.getEntry( name );
65          assertNotNull( ze );
66          InputStream is = null;
67          try {
68              is = zip.getInputStream( ze );
69              assertEquals(name, expected.replaceAll( "\\s","" ), IOUtil.toString( is ).trim().replaceAll( "\\s","" ));
70          }
71          finally
72          {
73              if( is != null ) is.close();
74          }
75      }
